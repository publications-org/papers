\begin{theorem}
    There exists in $\mathbb{R}$ the following identity:
    \begin{equation}
        \frac{\int (np) \text{ } dp}{\int (px) \text{ } dx} = \frac{p^2n+2C}{px^2+2C}
    \end{equation}
    which approaches zero for all elements of $\mathbb{R}$.
\end{theorem}
