\section{Theorems}
\begin{theorem}
  There exists a solution in $\mathbb{R}$:
  \begin{equation}
    \oint \frac{\ln(\frac{1}{n^s})}{\frac{1}{n^s}}ds = -n^s \cdot s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
  that has the following limit:
  \begin{equation}
    \lim _{x\to \infty } -n^s \cdot s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
  which when derived, evaluates to:
  \begin{equation}
    \lim _{x\to \infty } -n^s \cdot s+\frac{n^s}{\ln \left(n\right)}+C = \oint \frac{\ln(\frac{1}{n^s})}{\frac{1}{n^s}}ds
  \end{equation}
\end{theorem}

\begin{theorem}
  There exists in $\mathbb{R}$, for $2 < s < p$:
  \begin{equation}
    B = \lim _{n\:\to \infty }\:\sum_{n=1}^{p} \frac{1}{^p}
  \end{equation}
  which evaluates to the following:
  \begin{equation}
    t \pm \frac{1}{p}
  \end{equation}
  where $t > 0$ is the resulting integration of $p \in \mathbb{R}$.
\end{theorem}

\begin{theorem}
  There exists the following in $\mathbb{R}$:
  \begin{equation}
    C = \lim _{\left(k-p\right)\to \infty } \frac{(k-p)^{e}}{\ln(k-p)}
  \end{equation}
  for $(k-p) > 1$, for which [1.3], evaluates to:
  \begin{equation}
    \frac{(k-p)^{e}}{\ln(k-p)}
  \end{equation}
  when posing the following equation:
  \begin{equation}
    \lim _{\left(k-p\right)\to \infty } \frac{(k-p)^{e}}{\ln(k-p)} = \frac{(k-p)^{e}}{\ln(k-p)}
  \end{equation}
  where $e$ evaluates to Euler' constant.
\end{theorem}

\begin{theorem}
  Let the following intgeral with defined $z$, $x$, and $t$ values:
  \begin{equation}
    P(t, x, z) = \int \frac{\log_{10}|t \cdot z \cdot x|}{(z \cdot t)!} dx
  \end{equation}
  for any solutions of:
  \begin{equation}
    \:\:\pi< P(t, x, z) <\infty 
  \end{equation}
  [1.4] holds for $x, t \in \mathbb{R}, \quad \pi < t, x, z < \infty$.
\end{theorem}

\begin{theorem}
  There exists a following solution:
  \begin{equation}
    \lim _{s\to \:\pi }\left(-n^s+\frac{n^s}{\ln \left(n\right)}+C\right)
  \end{equation}
  for a function $F(s) \in \mathbb{R}$ that is greater than one and approaching $\pi$.
\end{theorem}
