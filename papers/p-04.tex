\section{Theorems}
\begin{theorem}
  There exists in $\mathbb{R}$ a solution:
  \begin{equation}
    h(n, s) \coloneqq \oint \frac{\ln(\frac{1}{n^s})}{\frac{1}{n^s}} \space ds \coloneq -n^s \cdot s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
  if the following limit converges towards $+\infty$, then:
  \begin{equation}
    h(n, s) \implies \lim _{x\to \infty } -n^s \cdot s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
\end{theorem}

\begin{theorem}
  There exists in $\mathbb{R}$, for $2 < s < p$:
  \begin{equation}
    z(p) \coloneq \lim _{n\:\to \infty }\:\sum_{n=1}^{p} \frac{1}{^p}
  \end{equation}
  s.t. If and only if:
  \begin{equation}
    z(p) \implies y \pm \frac{1}{p}
  \end{equation}
  for $y \in \mathbb{R}$, $y > 0$. If and only if, $y=z(p)$ in $p \in \mathbb{R}$.
\end{theorem}

\begin{theorem}
  There exists  in $\mathbb{R}$ the following:
  \begin{equation}
    c(p) \coloneq \frac{p!}{\ln \left(p\right)!}
  \end{equation}
  if and only if: $\ln(p)=p!$.
\end{theorem}

\begin{theorem}
  Let the following intgeral with the defined values: $z$, $x$, $y$:
  \begin{equation}
    \sigma(x, y, z) \coloneq \int \frac{f'(x \cdot y \cdot z)}{f''(x \cdot y \cdot z)} \space dt
  \end{equation}
  if for $x, y, z \in \mathbb{R}, \quad L < \sigma(x, y, z) < H$, we have solutions that converges towards $H$.
\end{theorem}

\begin{theorem}
  There exists a following solution:
  \begin{equation}
    f(s) \coloneq \lim _{s\to \:\pi } -n^s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
  if for a function $f(s) \in \mathbb{R}$, the following $f(s) > 1$, $f(s) \to \pi$ holds.
\end{theorem}

\begin{theorem}
  Let the product of an integral $\in \mathbb{R}$ with the defined with the following values from $a$, $\dots$, to $z$, with a converging function $\lambda(a, \dots, z)$:
  \begin{equation}
    \alpha(a, \dots, z) \coloneq \int \frac{\lambda(a, \dots, z)}{(a, \dots, z)!}! \space dx
  \end{equation}
  If and only if, $(a, \dots, z) > 1 \text{, and } (a, \dots, z) \in \mathbb{R}$ are met, then the limit of $f(a, \dots, z)$ converges towards:
  \begin{equation}
    \lim_{x} \alpha(a, \dots, z) = +\infty
  \end{equation}
\end{theorem}
