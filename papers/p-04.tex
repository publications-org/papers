\section{Theorems}
\begin{theorem}
  There exists in $\mathbb{R}$ a solution:
  \begin{equation}
    L(n, s) \coloneqq \oint \frac{\ln(\frac{1}{n^s})}{\frac{1}{n^s}}ds \coloneq -n^s \cdot s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
  with the following limit towards $\infty$:
  \begin{equation}
    L(n, s) \implies \lim _{x\to \infty } -n^s \cdot s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
\end{theorem}

\begin{theorem}
  There exists in $\mathbb{R}$, for $2 < s < p$:
  \begin{equation}
    B(p) \coloneq \lim _{n\:\to \infty }\:\sum_{n=1}^{p} \frac{1}{^p}
  \end{equation}
  s.t.:
  \begin{equation}
    B(p) \implies y \pm \frac{1}{p}
  \end{equation}
  for $y \in \mathbb{R}$, $y > 0$ when it is the resulting of an integration of $p \in \mathbb{R}$.
\end{theorem}

\begin{theorem}
  There exists  in $\mathbb{R}$ the following:
  \begin{equation}
    C(p) \coloneq \frac{p!}{\ln \left(p\right)!}
  \end{equation}
  thay may be solution that uses $\ln(p)$ for $ p! $ times.
\end{theorem}

\begin{theorem}
  Let the following intgeral with the defined values: $z$, $x$, $y$:
  \begin{equation}
    P(x, y, z) \coloneq \int \frac{P_1(x \cdot y \cdot z)}{P_2(x \cdot y \cdot z)} dx
  \end{equation}
  have for $x, y, z \in \mathbb{R}, \quad L < P(x, y, z) < H$, have solutions that converges towards $H$.
\end{theorem}

\begin{theorem}
  There exists a following solution:
  \begin{equation}
    \lim _{s\to \:\pi } -n^s+\frac{n^s}{\ln \left(n\right)}+C
  \end{equation}
  for a function $F(s) \in \mathbb{R}$ for the following $F(s) > 1$, and $F(s) \to \pi$.
\end{theorem}

\begin{theorem}
  Let the product of an integral $\in \mathbb{R}$ with the defined with the following values from $a$, $\dots$, to $z$:
  \begin{equation}
    F(a, \dots, z) \coloneq \int F_1(a, \dots, z) dz 
  \end{equation}
  if $(a, \dots, z) > 1 \text{, and } (a, \dots, z) \in \mathbb{R}$ are met, then the limit of $F(a, \dots, z)$ may converge to the following:
  \begin{equation}
    \lim_{x} F(a, \dots, z) = F_2(a, \dots, z) + C
  \end{equation}
\end{theorem}
