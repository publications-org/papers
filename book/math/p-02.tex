\section{Equations \& Theorems}
\begin{theorem}
    There exists:
    \begin{align*}
        \frac{\int (np) \text{ } \operatorname{dp}}{\int (px) \text{ } \operatorname{dx}} &= \frac{p^2n+2C}{px^2+2C} \in \mathbb{D}^{+};
    \end{align*}
    Therefore by dividing by $\int (px) \text{ } \operatorname{dx}$:
    \begin{align*}
        \int (np) \text{ } \operatorname{dp} &= \frac{p^2n+2C}{px^2+2C} \int (px) \text{ } \operatorname{dx}
    \end{align*}
\end{theorem}
