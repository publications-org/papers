\begin{definition}
	Let an equation of the following for all variables in $\mathbb{R}$:
	\begin{align*}
		&= \iiint (x y z)^{3} dx.dy.dz;\\
		c_{1} y z + c_{2} z + c_{3} + \frac{1}{64} + x^{4} + y^{4} + z^{4} &= \iiint (x y z)^{3} dx.dy.dz.
	\end{align*}
	An equation based on the result would be of the following:
	\begin{align*}
		(c_{1} y z + c_{2} z + c_{3} + \frac{1}{64} + x^{4} + y^{4} + z^{4})^{n-2} + \sum_{n=1}^{p} \frac{ (c_{1} y z + c_{2} z + c_{3} + \frac{1}{64} + x^{4} + y^{4} + z^{4})^{n}}{ (c_{1} y z + c_{2} z + c_{3} + \frac{1}{64} + x^{4} + y^{4} + z^{4})^{n-1}} &= \pm C;
	\end{align*}
\end{definition}
